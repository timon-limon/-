    # Список из целых чисел
    numbers = [1, 2, 3, 4, 5]
    print(numbers) # Вывод: [1, 2, 3, 4, 5]

    # Список из строк
    fruits = ['яблоко', 'банан', 'апельсин']
    print(fruits) # Вывод: ['яблоко', 'банан', 'апельсин']

    # Список с элементами разных типов
    my_list = [10, 'текст', 3.14, True]
    print(my_list) # Вывод: [10, 'текст', 3.14, True]

    # Пустой список
    empty_list = []
    print(empty_list) # Вывод: []
